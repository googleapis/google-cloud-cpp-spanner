130 StatusOr<CommitResult> RunTransactionImpl(
131  Client& client, Transaction::ReadWriteOptions const& opts,
132  std::function<StatusOr<Mutations>(Client, Transaction)> const& f) {
133  Transaction txn = MakeReadWriteTransaction(opts);
134  StatusOr<Mutations> mutations;
135 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
136  try {
137 #endif
138  mutations = f(client, txn);
139 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
140  } catch (...) {
141  auto status = client.Rollback(txn);
142  if (!status.ok()) {
143  GCP_LOG(WARNING) << "Rollback() failure in RunTransaction(): "
144  << status.message();
145  }
146  throw;
147  }
148 #endif
149  if (!mutations) {
150  auto status = client.Rollback(txn);
151  if (!status.ok()) {
152  GCP_LOG(WARNING) << "Rollback() failure in RunTransaction(): "
153  << status.message();
154  }
155  return mutations.status();
156  }
157  return client.Commit(txn, *mutations);
158 }

164 StatusOr<CommitResult> RunTransactionWithPolicies(
165  Client client, Transaction::ReadWriteOptions const& opts,
166  std::function<StatusOr<Mutations>(Client, Transaction)> const& f,
167  std::unique_ptr<TransactionRerunPolicy> rerun_policy,
168  std::unique_ptr<BackoffPolicy> backoff_policy) {
169  Status last_status(
170  StatusCode::kFailedPrecondition,
171  "Retry policy should not be exhausted when retry loop starts");
172  char const* reason = "Too many failures in ";
173  while (!rerun_policy->IsExhausted()) {
174  auto result = RunTransactionImpl(client, opts, f);
175  if (result) return result;
176  last_status = std::move(result).status();
177  if (!rerun_policy->OnFailure(last_status)) {
178  if (internal::SafeTransactionRerun::IsPermanentFailure(last_status)) {
179  reason = "Permanent failure in ";
180  }
181  break;
182  }
183  std::this_thread::sleep_for(backoff_policy->OnCompletion());
184  }
185  return internal::RetryLoopError(reason, __func__, last_status);
186 }
