57 grpc::ChannelArguments ConnectionOptions::CreateChannelArguments() const {
58  grpc::ChannelArguments channel_arguments;
59 
60  if (!channel_pool_domain().empty()) {
61  // To get a different channel pool one just needs to set any channel
62  // parameter to a different value. Newer versions of gRPC include a macro
63  // for this purpose (GRPC_ARG_CHANNEL_POOL_DOMAIN). As we are compiling
64  // against older versions in some cases, we use the actual value.
65  channel_arguments.SetString("grpc.channel_pooling_domain",
66  channel_pool_domain());
67  }
68  channel_arguments.SetUserAgentPrefix(user_agent_prefix());
69  return channel_arguments;
70 }
