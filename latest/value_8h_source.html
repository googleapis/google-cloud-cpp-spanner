339  template <typename T>
340  static google::spanner::v1::Type MakeTypeProto(std::vector<T> const& v) {
341  google::spanner::v1::Type t;
342  t.set_code(google::spanner::v1::TypeCode::ARRAY);
343  *t.mutable_array_element_type() = MakeTypeProto(v.empty() ? T{} : v[0]);
344  // Checks that vector elements have exactly the same proto Type, which
345  // includes field names. This is documented UB.
346  for (auto const& e : v) {
347  if (!google::protobuf::util::MessageDifferencer::Equals(
348  MakeTypeProto(e), t.array_element_type())) {
349  google::cloud::internal::ThrowInvalidArgument("Mismatched types");
350  }
351  }
352  return t;
353  }

356  google::spanner::v1::Type t;

465  // A functor to be used with IterateTuple (see below) to extract C++ types
466  // from a ListValue proto and store then in a tuple.
467  struct ExtractTupleValues {
468  std::size_t i;
469  google::protobuf::ListValue const& list_value;
470  google::spanner::v1::Type const& type;

484  // A helper to iterate the elements of the tuple, calling the given functor
485  // `f` with each tuple element and any optional `args`. Typically `F` should
486  // be a functor type with a templated operator() so that it can handle the
487  // various types in the tuple.
488  template <std::size_t I = 0, typename Tup, typename F, typename... Args>
489  static typename std::enable_if<
490  (I < std::tuple_size<typename std::decay<Tup>::type>::value), void>::type
491  IterateTuple(Tup&& tup, F&& f, Args&&... args) {
492  auto&& e = std::get<I>(std::forward<Tup>(tup));
493  std::forward<F>(f)(std::forward<decltype(e)>(e),
494  std::forward<Args>(args)...);
495  IterateTuple<I + 1>(std::forward<Tup>(tup), std::forward<F>(f),
496  std::forward<Args>(args)...);
497  }
498  template <std::size_t I = 0, typename Tup, typename F, typename... Args>
499  static typename std::enable_if<
500  I == std::tuple_size<typename std::decay<Tup>::type>::value, void>::type
501  IterateTuple(Tup&&, F&&, Args&&...) {}
