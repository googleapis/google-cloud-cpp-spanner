425  template <typename T>
426  static google::spanner::v1::Type MakeTypeProto(std::vector<T> const& v) {
427  google::spanner::v1::Type t;
428  t.set_code(google::spanner::v1::TypeCode::ARRAY);
429  *t.mutable_array_element_type() = MakeTypeProto(v.empty() ? T{} : v[0]);
430  // Checks that vector elements have exactly the same proto Type, which
431  // includes field names. This is documented UB.
432  for (auto const& e : v) {
433  if (!google::protobuf::util::MessageDifferencer::Equals(
434  MakeTypeProto(e), t.array_element_type())) {
435  google::cloud::internal::ThrowInvalidArgument("Mismatched types");
436  }
437  }
438  return t;
439  }

536  template <typename T>
537  static StatusOr<optional<T>> GetValue(optional<T> const&,
538  google::protobuf::Value const& pv,
539  google::spanner::v1::Type const& pt) {
540  if (pv.kind_case() == google::protobuf::Value::kNullValue) {
541  return optional<T>{};
542  }
543  auto value = GetValue(T{}, pv, pt);
544  if (!value) return std::move(value).status();
545  return optional<T>{*std::move(value)};
546  }
547  template <typename T>
548  static StatusOr<std::vector<T>> GetValue(
549  std::vector<T> const&, google::protobuf::Value const& pv,
550  google::spanner::v1::Type const& pt) {
551  if (pv.kind_case() != google::protobuf::Value::kListValue) {
552  return Status(StatusCode::kUnknown, "missing ARRAY");
553  }
554  std::vector<T> v;
555  for (auto const& e : pv.list_value().values()) {
556  auto value = GetValue(T{}, e, pt.array_element_type());
557  if (!value) return std::move(value).status();
558  v.push_back(*std::move(value));
559  }
560  return v;
561  }
562  template <typename... Ts>
563  static StatusOr<std::tuple<Ts...>> GetValue(
564  std::tuple<Ts...> const&, google::protobuf::Value const& pv,
565  google::spanner::v1::Type const& pt) {
566  if (pv.kind_case() != google::protobuf::Value::kListValue) {
567  return Status(StatusCode::kUnknown, "missing STRUCT");
568  }
569  std::tuple<Ts...> tup;
570  Status status; // OK
571  ExtractTupleValues f{status, 0, pv.list_value(), pt};
572  internal::ForEach(tup, f);
573  if (!status.ok()) return status;
574  return tup;
575  }

579  struct ExtractTupleValues {

581  std::size_t i;
