495  template <typename T, typename V>
496  static StatusOr<optional<T>> GetValue(optional<T> const&, V&& pv,
497  google::spanner::v1::Type const& pt) {
498  if (pv.kind_case() == google::protobuf::Value::kNullValue) {
499  return optional<T>{};
500  }
501  auto value = GetValue(T{}, std::forward<V>(pv), pt);
502  if (!value) return std::move(value).status();
503  return optional<T>{*std::move(value)};
504  }
505  template <typename T, typename V>
506  static StatusOr<std::vector<T>> GetValue(
507  std::vector<T> const&, V&& pv, google::spanner::v1::Type const& pt) {
508  if (pv.kind_case() != google::protobuf::Value::kListValue) {
509  return Status(StatusCode::kUnknown, "missing ARRAY");
510  }
511  std::vector<T> v;
512  for (int i = 0; i < pv.list_value().values().size(); ++i) {
513  auto&& e = GetProtoListValueElement(std::forward<V>(pv), i);
514  using ET = decltype(e);
515  auto value = GetValue(T{}, std::forward<ET>(e), pt.array_element_type());
516  if (!value) return std::move(value).status();
517  v.push_back(*std::move(value));
518  }
519  return v;
520  }
521  template <typename V, typename... Ts>
522  static StatusOr<std::tuple<Ts...>> GetValue(
523  std::tuple<Ts...> const&, V&& pv, google::spanner::v1::Type const& pt) {
524  if (pv.kind_case() != google::protobuf::Value::kListValue) {
525  return Status(StatusCode::kUnknown, "missing STRUCT");
526  }
527  std::tuple<Ts...> tup;
528  Status status; // OK
529  ExtractTupleValues<V> f{status, 0, std::forward<V>(pv), pt};
530  internal::ForEach(tup, f);
531  if (!status.ok()) return status;
532  return tup;
533  }
