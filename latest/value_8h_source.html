368  google::spanner::v1::Type t;
369  t.set_code(google::spanner::v1::TypeCode::ARRAY);

371  // Checks that vector elements have exactly the same proto Type, which

477  template <typename T>
478  static StatusOr<optional<T>> GetValue(optional<T> const&,
479  google::protobuf::Value const& pv,
480  google::spanner::v1::Type const& pt) {
481  if (pv.kind_case() == google::protobuf::Value::kNullValue) {
482  return optional<T>{};
483  }
484  auto value = GetValue(T{}, pv, pt);
485  if (!value) return std::move(value).status();
486  return optional<T>{*std::move(value)};
487  }
488  template <typename T>
489  static StatusOr<std::vector<T>> GetValue(
490  std::vector<T> const&, google::protobuf::Value const& pv,
491  google::spanner::v1::Type const& pt) {
492  if (pv.kind_case() != google::protobuf::Value::kListValue) {
493  return Status(StatusCode::kUnknown, "missing ARRAY");
494  }
495  std::vector<T> v;
496  for (auto const& e : pv.list_value().values()) {
497  auto value = GetValue(T{}, e, pt.array_element_type());
498  if (!value) return std::move(value).status();
499  v.push_back(*std::move(value));
500  }
501  return v;
502  }
503  template <typename... Ts>
504  static StatusOr<std::tuple<Ts...>> GetValue(
505  std::tuple<Ts...> const&, google::protobuf::Value const& pv,
506  google::spanner::v1::Type const& pt) {
507  if (pv.kind_case() != google::protobuf::Value::kListValue) {
508  return Status(StatusCode::kUnknown, "missing STRUCT");
509  }
510  std::tuple<Ts...> tup;
511  Status status; // OK
512  ExtractTupleValues f{status, 0, pv.list_value(), pt};
513  internal::ForEach(tup, f);
514  if (!status.ok()) return status;
515  return tup;
516  }
