24 std::string RandomDatabaseName(
25  google::cloud::internal::DefaultPRNG& generator) {
26  // A database ID must be between 2 and 30 characters, fitting the regular
27  // expression `[a-z][a-z0-9_\-]*[a-z0-9]`
28  int max_size = 30;
29  std::string const prefix = "db-";
30  auto size = static_cast<int>(max_size - 1 - prefix.size());
31  return prefix +
32  google::cloud::internal::Sample(
33  generator, size, "abcdefghijlkmnopqrstuvwxyz012345689_-") +
34  google::cloud::internal::Sample(generator, 1,
35  "abcdefghijlkmnopqrstuvwxyz");
36 }

38 void CreateDatabase(std::vector<std::string> const& argv) {
39  if (argv.size() != 3) {
40  throw std::runtime_error(
41  "create-database <project-id> <instance-id> <database-id>");
42  }
43 
45  using google::cloud::future;
46  using google::cloud::StatusOr;
47  [](std::string const& project_id, std::string const& instance_id,
48  std::string const& database_id) {
49  google::cloud::spanner::DatabaseAdminClient client;
50  future<StatusOr<google::spanner::admin::database::v1::Database>> future =
51  client.CreateDatabase(project_id, instance_id, database_id,
52  {R"""(
53  CREATE TABLE Singers (
54  SingerId INT64 NOT NULL,
55  FirstName STRING(1024),
56  LastName STRING(1024),
57  SingerInfo BYTES(MAX)
58  ) PRIMARY KEY (SingerId))""",
59  R"""(CREATE TABLE Albums (
60  SingerId INT64 NOT NULL,
61  AlbumId INT64 NOT NULL,
62  AlbumTitle STRING(MAX)
63  ) PRIMARY KEY (SingerId, AlbumId),
64  INTERLEAVE IN PARENT Singers ON DELETE CASCADE)"""});
65  StatusOr<google::spanner::admin::database::v1::Database> db = future.get();
66  if (!db) {
67  throw std::runtime_error(db.status().message());
68  }
69  std::cout << "Created database [" << database_id << "]\n";
70  }
72  (argv[0], argv[1], argv[2]);
73 }

137 int RunOneCommand(std::vector<std::string> argv) {
138  using CommandType = std::function<void(std::vector<std::string> const&)>;
139 
140  std::map<std::string, CommandType> commands = {
141  {"create-database", &CreateDatabase},
142  {"add-column", &AddColumn},
143  {"query-with-struct", &QueryWithStruct},
144  {"drop-database", &DropDatabase},
145  };
146 
147  std::string usage_msg = [&argv, &commands] {
148  auto last_slash = std::string(argv[0]).find_last_of("/\\");
149  auto program = argv[0].substr(last_slash + 1);
150  std::string usage;
151  usage += "Usage: ";
152  usage += program;
153  usage += " <command> [arguments]\n\n";
154  usage += "Commands:\n";
155  for (auto&& kv : commands) {
156  try {
157  kv.second({});
158  } catch (std::exception const& ex) {
159  usage += " ";
160  usage += ex.what();
161  usage += "\n";
162  }
163  }
164  return usage;
165  }();
166 
167  if (argv.size() < 2) {
168  std::cerr << "Missing command argument\n" << usage_msg << "\n";
169  return 1;
170  }
171  std::string command_name = argv[1];
172  argv.erase(argv.begin()); // remove the program name from the list.
173  argv.erase(argv.begin()); // remove the command name from the list.
174 
175  auto command = commands.find(command_name);
176  if (commands.end() == command) {
177  std::cerr << "Unknown command " << command_name << "\n"
178  << usage_msg << "\n";
179  return 1;
180  }
181 
182  // Run the command.
183  command->second(argv);
184  return 0;
185 }

187 void RunAll() {
188  auto project_id =
189  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value_or("");
190  auto instance_id =
191  google::cloud::internal::GetEnv("GOOGLE_CLOUD_CPP_SPANNER_INSTANCE")
192  .value_or("");
193 
194  if (project_id.empty()) {
195  throw std::runtime_error("GOOGLE_CLOUD_PROJECT is not set or is empty");
196  }
197  if (instance_id.empty()) {
198  throw std::runtime_error(
199  "GOOGLE_CLOUD_CPP_SPANNER_INSTANCE is not set or is empty");
200  }
201  auto generator = google::cloud::internal::MakeDefaultPRNG();
202  std::string database_id = RandomDatabaseName(generator);
203 
204  RunOneCommand({"", "create-database", project_id, instance_id, database_id});
205  RunOneCommand({"", "add-column", project_id, instance_id, database_id});
206  RunOneCommand(
207  {"", "query-with-struct", project_id, instance_id, database_id});
208  RunOneCommand({"", "drop-database", project_id, instance_id, database_id});
209 }
