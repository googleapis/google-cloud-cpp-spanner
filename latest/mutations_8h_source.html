100 template <typename Op>
101 class WriteMutationBuilder {
102  public:
103  WriteMutationBuilder(std::string table_name,
104  std::vector<std::string> column_names) {
105  auto& field = Op::mutable_field(m_);
106  field.set_table(std::move(table_name));
107  field.mutable_columns()->Reserve(static_cast<int>(column_names.size()));
108  for (auto& name : column_names) {
109  *field.add_columns() = std::move(name);
110  }
111  }
112 
113  Mutation Build() const& { return Mutation(m_); }
114  Mutation Build() && { return Mutation(std::move(m_)); }
115 
116  WriteMutationBuilder& AddRow(std::vector<Value> values) {
117  auto& lv = *Op::mutable_field(m_).add_values();
118  for (auto& v : values) {
119  std::tie(std::ignore, *lv.add_values()) = internal::ToProto(std::move(v));
120  }
121  return *this;
122  }
123 
124  template <typename... Ts>
125  WriteMutationBuilder& EmplaceRow(Ts&&... values) {
126  return AddRow({Value(std::forward<Ts>(values))...});
127  }
128 
129  private:
130  google::spanner::v1::Mutation m_;
131 };
