92 template <typename Op>
93 class WriteMutationBuilder {
94  public:
95  WriteMutationBuilder(std::string table_name,
96  std::vector<std::string> column_names) {
97  auto& field = Op::mutable_field(m_);
98  field.set_table(std::move(table_name));
99  field.mutable_columns()->Reserve(column_names.size());
100  for (auto& name : column_names) {
101  *field.add_columns() = std::move(name);
102  }
103  }
104 
105  Mutation Build() const& { return Mutation(m_); }
106  Mutation Build() && { return Mutation(std::move(m_)); }
107 
108  template <typename... Ts>
109  WriteMutationBuilder& AddRow(Row<Ts...> row) {
110  auto& lv = *Op::mutable_field(m_).add_values();
111  for (auto& v : std::move(row).values()) {
112  std::tie(std::ignore, *lv.add_values()) = internal::ToProto(std::move(v));
113  }
114  return *this;
115  }
116 
117  template <typename... Ts>
118  WriteMutationBuilder& EmplaceRow(Ts&&... values) {
119  return AddRow(MakeRow(std::forward<Ts>(values)...));
120  }
121 
122  private:
123  google::spanner::v1::Mutation m_;
124 };
