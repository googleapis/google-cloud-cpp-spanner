164  // A helper used with `internal::ForEach` to convert an array of `Value`
165  // objects into a std::tuple.
166  struct ExtractValue {
167  Status& status;
168  template <typename T, typename It>
169  void operator()(T& t, It& it) const {
170  auto x = it++->template get<T>();
171  if (!x) {
172  status = std::move(x).status();
173  } else {
174  t = *std::move(x);
175  }
176  }

179  template <std::size_t N>
180  StatusOr<TupleType> ParseRow(std::array<Value, N> const& array) {
181  TupleType row;
182  auto it = array.begin();
183  Status status;
184  internal::ForEach(row, ExtractValue{status}, it);
185  if (!status.ok()) return status;
186  return row;
187  }
188 
189  // Consumes Values from value_source_ and stores the consumed std::tuple in
190  // curr_. Called by iterator::operator++().
191  void Advance() {
192  if (curr_ && !*curr_) { // Last row was an error; jump to end
193  value_source_ = nullptr;
194  return;
195  }
196  std::array<Value, std::tuple_size<TupleType>::value> values;
197  for (std::size_t i = 0; i < values.size(); ++i) {
198  StatusOr<optional<Value>> v = value_source_();
199  if (!v) {
200  curr_ = std::move(v).status();
201  return;
202  }
203  if (!*v) {
204  if (i == 0) { // We've successfully reached the end.
205  value_source_ = nullptr;
206  } else {
207  curr_ = Status(StatusCode::kUnknown, "incomplete row");
208  }
209  return;
210  }
211  values[i] = **std::move(v);
212  }
213  curr_ = ParseRow(values);
214  }

216  ValueSource value_source_; // nullpr is end
217  optional<value_type> curr_;
