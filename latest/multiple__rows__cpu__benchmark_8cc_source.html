133  generator, config.project_id);

144  google::cloud::spanner_testing::RandomDatabaseName(generator));

148  // print everything out.

371  std::string const& table_name) {

378  int const task_count = 16;

470  private:
471  Status FillTableTask(Config const& config, cs::Client client,
472  std::string const& table_name, int task_count,
473  int task_id) {
474  std::vector<std::string> const column_names{
475  "Key", "Data0", "Data1", "Data2", "Data3", "Data4",
476  "Data5", "Data6", "Data7", "Data8", "Data9"};
477  using T = typename Traits::native_type;
478  T value0 = GenerateRandomValue();
479  T value1 = GenerateRandomValue();
480  T value2 = GenerateRandomValue();
481  T value3 = GenerateRandomValue();
482  T value4 = GenerateRandomValue();
483  T value5 = GenerateRandomValue();
484  T value6 = GenerateRandomValue();
485  T value7 = GenerateRandomValue();
486  T value8 = GenerateRandomValue();
487  T value9 = GenerateRandomValue();
488 
489  auto mutation = cs::InsertOrUpdateMutationBuilder(table_name, column_names);
490  int current_mutations = 0;
491 
492  auto maybe_flush = [&, this](bool force) -> Status {
493  if (current_mutations == 0) {
494  return {};
495  }
496  if (!force && current_mutations < 1000) {
497  return {};
498  }
499  auto m = std::move(mutation).Build();
500  auto result = client.Commit(
501  [&m](cs::Transaction const&) { return cs::Mutations{m}; });
502  if (!result) {
503  std::lock_guard<std::mutex> lk(mu_);
504  std::cerr << "# Error in Commit() " << result.status() << "\n";
505  return std::move(result).status();
506  }
507  mutation = cs::InsertOrUpdateMutationBuilder(table_name, column_names);
508  current_mutations = 0;
509  return {};
510  };
511  auto force_flush = [&maybe_flush] { return maybe_flush(true); };
512  auto flush_as_needed = [&maybe_flush] { return maybe_flush(false); };
513 
514  auto const report_period =
515  (std::max)(static_cast<std::int64_t>(2), config.table_size / 50);
516  for (std::int64_t key = 0; key != config.table_size; ++key) {
517  // Each thread does a fraction of the key space.
518  if (key % task_count != task_id) continue;
519  // Have one of the threads report progress about 50 times.
520  if (task_id == 0 && key % report_period == 0) {
521  std::cout << '.' << std::flush;
522  }
523  mutation.EmplaceRow(key, value0, value1, value2, value3, value4, value5,
524  value6, value7, value8, value9);
525  ++current_mutations;
526  auto status = flush_as_needed();
527  if (!status.ok()) return status;
528  }
529  return force_flush();
530  }

550 class ReadExperiment : public Experiment {
551  public:
552  explicit ReadExperiment(google::cloud::internal::DefaultPRNG generator)
553  : impl_(generator),
554  table_name_("ReadExperiment_" + Traits::TableSuffix()) {}

556  Status SetUp(Config const& config, cs::Database const& database) override {
557  return impl_.FillTable(config, database, table_name_);
558  }

560  Status TearDown(Config const&, cs::Database const&) override { return {}; }

562  Status Run(Config const& config, cs::Database const& database) override {
563  // Create enough clients and stubs for the worst case
564  std::vector<cs::Client> clients;
565  std::vector<std::shared_ptr<cs::internal::SpannerStub>> stubs;
566  std::cout << "# Creating clients and stubs " << std::flush;
567  for (int i = 0; i != config.maximum_clients; ++i) {
568  auto options = cs::ConnectionOptions().set_channel_pool_domain(
569  "task:" + std::to_string(i));
570  clients.emplace_back(cs::Client(cs::MakeConnection(database, options)));
571  stubs.emplace_back(
572  cs::internal::CreateDefaultSpannerStub(options, /*channel_id=*/0));
573  std::cout << '.' << std::flush;
574  }
575  std::cout << " DONE\n";
576 
577  // Capture some overall getrusage() statistics as comments.
578  SimpleTimer overall;
579  overall.Start();
580  for (int i = 0; i != config.samples; ++i) {
581  auto const use_stubs = impl_.UseStub(config);
582  auto const thread_count = impl_.ThreadCount(config);
583  auto const client_count = impl_.ClientCount(config, thread_count);
584  if (use_stubs) {
585  std::vector<std::shared_ptr<cs::internal::SpannerStub>> iteration_stubs(
586  stubs.begin(), stubs.begin() + client_count);
587  RunIterationViaStubs(config, iteration_stubs, thread_count);
588  continue;
589  }
590  std::vector<cs::Client> iteration_clients(clients.begin(),
591  clients.begin() + client_count);
592  RunIterationViaClients(config, iteration_clients, thread_count);
593  }
594  overall.Stop();
595  std::cout << overall.annotations();
596  return {};
597  }

599  private:
600  void RunIterationViaStubs(
601  Config const& config,
602  std::vector<std::shared_ptr<cs::internal::SpannerStub>> const& stubs,
603  int thread_count) {
604  std::vector<std::future<std::vector<RowCpuSample>>> tasks(thread_count);
605  int task_id = 0;
606  for (auto& t : tasks) {
607  auto client = stubs[task_id++ % stubs.size()];
608  t = std::async(std::launch::async, &ReadExperiment::ReadRowsViaStub, this,
609  config, thread_count, static_cast<int>(stubs.size()),
610  cs::Database(config.project_id, config.instance_id,
611  config.database_id),
612  client);
613  }
614  for (auto& t : tasks) {
615  impl_.DumpSamples(t.get());
616  }
617  }

619  std::vector<RowCpuSample> ReadRowsViaStub(
620  Config const& config, int thread_count, int client_count,
621  cs::Database const& database,
622  std::shared_ptr<cs::internal::SpannerStub> const& stub) {
623  auto session = [&]() -> google::cloud::StatusOr<std::string> {
624  Status last_status;
625  for (int i = 0; i != 10; ++i) {
626  grpc::ClientContext context;
627  google::spanner::v1::CreateSessionRequest request{};
628  request.set_database(database.FullName());
629  auto response = stub->CreateSession(context, request);
630  if (response) return response->name();
631  last_status = response.status();
632  }
633  return last_status;
634  }();
635 
636  if (!session) {
637  std::ostringstream os;
638  os << "SESSION ERROR = " << session.status();
639  impl_.LogError(os.str());
640  return {};
641  }
642 
643  std::vector<RowCpuSample> samples;
644  // We expect about 50 reads per second per thread. Use that to estimate
645  // the size of the vector.
646  samples.reserve(config.iteration_duration.count() * 50);
647  std::vector<std::string> const columns{"Key", "Data0", "Data1", "Data2",
648  "Data3", "Data4", "Data5", "Data6",
649  "Data7", "Data8", "Data9"};
650  for (auto start = std::chrono::steady_clock::now(),
651  deadline = start + config.iteration_duration;
652  start < deadline; start = std::chrono::steady_clock::now()) {
653  auto key = impl_.RandomKeySet(config);
654 
655  SimpleTimer timer;
656  timer.Start();
